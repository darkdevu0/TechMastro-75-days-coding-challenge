class Solution {
public:
    vector<vector<int>> generate(int numRows) {
        vector<vector<int>> res;
        
        for(int i = 0; i < numRows; ++i) {
            vector<int> temp;
            
            for(int j = 0; j <= i; ++j) {
                if(j == 0 or j == i) temp.push_back(1);
                else temp.push_back(res.back()[j] + res.back()[j - 1]);
            }
            
            res.push_back(temp);
        }
        
        return res;
    }
};
